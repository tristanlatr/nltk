@staticmethod
def multiply(discourse, readings): (source)

Multiply every thread in discourse by every reading in readings.

Given discourse = [['A'], ['B']], readings = ['a', 'b', 'c'] , returns [['A', 'a'], ['A', 'b'], ['A', 'c'], ['B', 'a'], ['B', 'b'], ['B', 'c']]
